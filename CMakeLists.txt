cmake_minimum_required(VERSION 3.15)
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CXX_EXTENSIONS OFF)
enable_testing()

project(cpp-template)
add_library(cpp-template INTERFACE)
target_include_directories(cpp-template INTERFACE include)
add_library(thezhe::cpp-template ALIAS cpp-template)
add_executable(CTest CTest.cpp)
target_link_libraries(CTest thezhe::cpp-template)
add_test(CTest CTest)
